<script>
let cart=[],total=0,delivery=5;
function q(b,d){let s=b.parentElement.querySelector('span');let v=+s.innerText+d;if(v<1)v=1;s.innerText=v}
function add(btn,nome){const info=btn.parentElement;let base=+info.querySelector('.price').dataset.price;let qty=+info.querySelector('.qty span').innerText;let extraTxt='';info.querySelectorAll('input:checked').forEach(e=>{let [n,p]=e.value.split('|');base+=+p;extraTxt+=` + ${n}`});cart.push(`${qty}x ${nome}${extraTxt}`);total+=base*qty;render()}

function checkStatus(){let h=new Date().getHours();let s=document.getElementById('status');if(h>=8&&h<=18){s.innerText='ABERTO';s.className='status open'}else{s.innerText='FECHADO';s.className='status closed'}}
checkStatus();
